## Overview

Direct Messages (DMs) provide private one-on-one conversations between users. Unlike channels, DMs are routed through a dedicated user channel and use a conversation-based model with automatic thread management.

## DM Architecture

### User Channel

All DMs are routed through a single user-specific channel:

```typescript
// From channel-manager.ts:154
async joinUserChannel(userId: number): Promise<void> {
  if (!this.socket) {
    throw new Error("Socket not connected. Call connect() first.");
  }
  
  this.userId = userId;
  
  return new Promise((resolve, reject) => {
    this.userChannel = this.socket!.channel(`user:${userId}`, {});
    
    // Setup DM event handlers
    this.setupUserChannelHandlers();
    
    this.userChannel.join()
      .receive("ok", () => resolve())
      .receive("error", (error) => reject(error));
  });
}
```

<Info>
The user channel must be joined after connecting the socket and before sending or receiving DMs.
</Info>

### DM Slug Format

DM conversations use slugs in the format `dm:{user1_id}:{user2_id}`:

```typescript
interface DmConversation {
  slug: string;                    // e.g., "dm:123:456"
  other_user_id: number;
  other_username: string;
  last_message_preview: string;    // Truncated to 100 chars
  last_activity_at: string;        // ISO 8601 timestamp
  unread_count: number;
}
```

## Creating DM Conversations

### Create or Get Existing

```typescript
// From chat-client.ts:249
export async function createOrGetDm(
  wsUrl: string,
  token: string,
  otherUser: { user_id: number } | { username: string }
): Promise<CreateDmResponse> {
  const backendUrl = wsUrl
    .replace(/^wss:/, "https:")
    .replace(/^ws:/, "http:")
    .replace(/\/socket$/, "");
  
  const url = `${backendUrl}/api/dm`;
  
  const body = "user_id" in otherUser
    ? { other_user_id: otherUser.user_id }
    : { other_username: otherUser.username };
  
  const response = await fetch(url, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${token}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify(body)
  });
  
  if (!response.ok) {
    const data = await response.json();
    throw new Error(data.error || `Failed to create DM: ${response.status}`);
  }
  
  return response.json();
}
```

### Sorting Conversations

Sort by most recent activity:

```typescript
// From dm-utils.ts:22
export function sortConversationsByActivity(
  conversations: DmConversation[]
): DmConversation[] {
  return [...conversations].sort((a, b) => {
    return new Date(b.last_activity_at).getTime() - 
           new Date(a.last_activity_at).getTime();
  });
}
```

## Fetching DM Messages

Retrieve message history for a specific conversation:

```typescript
// From chat-client.ts:285
export async function fetchDmMessages(
  wsUrl: string,
  token: string,
  dmSlug: string,
  limit: number = 50
): Promise<{ messages: Message[]; dm_slug: string }> {
  const backendUrl = wsUrl
    .replace(/^wss:/, "https:")
    .replace(/^ws:/, "http:")
    .replace(/\/socket$/, "");
  
  const encodedSlug = encodeURIComponent(dmSlug);
  const url = `${backendUrl}/api/dm/${encodedSlug}/messages?limit=${limit}`;
  
  const response = await fetch(url, {
    headers: { Authorization: `Bearer ${token}` }
  });
  
  if (!response.ok) {
    throw new Error(`Failed to fetch DM messages: ${response.status}`);
  }
  
  return response.json();
}
```

## Sending DM Messages

### Send via User Channel

```typescript
// From channel-manager.ts:912
async sendDmMessage(
  dmSlug: string,
  content: string,
  attributes?: MessageAttributes,
  type?: "cc"
): Promise<{ message_id: string }> {
  if (!this.userChannel) {
    throw new Error("User channel not connected");
  }
  
  if (!this.socket || this.connectionStatus !== "connected") {
    throw new Error("Connection lost");
  }
  
  const payload = {
    dm_slug: dmSlug,
    content,
    attributes,
    type
  };
  
  return new Promise((resolve, reject) => {
    this.userChannel!.push("dm:send", payload)
      .receive("ok", (resp) => resolve(resp))
      .receive("error", (err) => reject(err));
  });
}
```

### DM Message Structure

```typescript
interface DmMessage {
  id: string;              // UUIDv7
  dm_slug: string;         // Conversation identifier
  username: string;        // Sender username
  content: string;         // Message content
  attributes?: MessageAttributes;
}
```

### Timestamp Extraction

DM messages use the same UUIDv7 timestamp extraction:

```typescript
// From DmChatView.tsx:31
function extractTimestampFromUUIDv7(uuid: string): string {
  const hex = uuid.replace(/-/g, "").slice(0, 12);
  const ms = parseInt(hex, 16);
  return new Date(ms).toISOString();
}

// Convert DmMessage to Message
const message: Message = {
  id: msg.id,
  username: msg.username,
  content: msg.content,
  timestamp: extractTimestampFromUUIDv7(msg.id),
  attributes: msg.attributes
};
```

## Best Practices

<AccordionGroup>
  <Accordion title="Connection Setup">
    - Join user channel immediately after socket connection
    - Ensure user channel is ready before sending DMs
    - Handle user channel disconnection gracefully
  </Accordion>
  
  <Accordion title="Conversation Management">
    - Use `createOrGetDm()` for idempotent conversation creation
    - Sort conversations by `last_activity_at` for inbox view
    - Truncate previews to 100 characters consistently
  </Accordion>
  
  <Accordion title="Message Handling">
    - Filter incoming DMs by `dm_slug` in callbacks
    - Extract timestamps from UUIDv7 message IDs
    - Mark as read both on entry and cleanup
  </Accordion>
  
  <Accordion title="Typing Indicators">
    - Always include `dm_slug` in typing events
    - Filter typing events by active conversation
    - Silently ignore typing indicator errors
  </Accordion>
</AccordionGroup>
